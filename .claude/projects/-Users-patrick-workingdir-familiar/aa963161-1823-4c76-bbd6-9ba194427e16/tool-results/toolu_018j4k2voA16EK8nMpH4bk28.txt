     4→// Mini sparkline component
     5→const Sparkline = ({ data, color = '#8b5cf6', height = 20, width = 60 }) => {
     6→  const max = Math.max(...data);
     7→  const min = Math.min(...data);
     8→  const range = max - min || 1;
     9→  
    10→  const points = data.map((val, i) => {
    11→    const x = (i / (data.length - 1)) * width;
    12→    const y = height - ((val - min) / range) * height;
    13→    return `${x},${y}`;
    14→  }).join(' ');
    15→  
    16→  return (
    17→    <svg width={width} height={height} style={{ display: 'block' }}>
    18→      <polyline
    19→        points={points}
    20→        fill="none"
    21→        stroke={color}
    22→        strokeWidth="1.5"
    23→        strokeLinecap="round"
    24→        strokeLinejoin="round"
    25→      />
    26→    </svg>
    27→  );
    28→};
